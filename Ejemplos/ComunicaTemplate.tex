/**
	 * \brief Instancia un nuevo usuario.
	 *
	 * @param id the id
	 * @param nombre el nombre de pila del usuario.
	 * @param apellidos el apellido del usuario.
	 * @param string 
	 * @param rol el rol de usuario
	 * @param mail the mail
	 */
	public Usuario(Long id, String nombreUsuario, String nombre,
								 String apellidos, Rol rol, String mail) {

		this.nombre = nombre;
		this.apellidos = apellidos;
		this.rol = rol;
		this.email=mail;
		this.id=id;
		this.nombreUsuario = nombreUsuario;
	}
